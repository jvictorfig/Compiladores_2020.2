	Programa que exibe todos os números pares e ímpares de 1 a 100





}

 verificaPareImpar() start
 	integer num<<1;
	while(num<=100)start
  		if(num%2=0)start
			show('Número par : '|num);
		end
		else start
			show('Número ímpar : '|num);
		end
        	+ num 1 ;
   	end 
	send back;
end

none main()start

	verificaPareImpar()	
 	send back;
  	
end
