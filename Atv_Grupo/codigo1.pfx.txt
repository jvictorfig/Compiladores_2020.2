	Programa que exibe todos os números pares e ímpares de 1 a 100





}

 verificaPareImpar() start
 	integer num<<1;
	while(num<=100)start
  		if(num%2=0)start
			printf("Número par : %d \n ",num);
		end
		else start
			printf("Número ímpar : %d \n ",num);
		end
        	++num;
   	end 
	send back;
end

none main()start

	verificaPareImpar();	
 	send back;
  	
end
